package com.comit.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.reactive.CorsWebFilter;
import org.springframework.web.cors.reactive.UrlBasedCorsConfigurationSource;
import org.springframework.web.util.pattern.PathPatternParser;

/**
 * @program: science-innovation-parent
 * @description: 跨域资源配置类，配置网关全局CORS策略，允许所有来源、方法和请求头
 * @author: comit
 * @create: 2020/01/01
 */
@Configuration
public class CorsConfig {

    /**
     * 配置跨域过滤器
     * @return CorsWebFilter 跨域过滤器实例
     */
    // TODO: 当前允许所有来源(*)访问，存在安全风险，生产环境应限制为具体域名
    @Bean
    public CorsWebFilter corsFilter() {
        CorsConfiguration config = new CorsConfiguration();
        config.addAllowedMethod("*");
        config.addAllowedOrigin("*");
        config.addAllowedHeader("*");

        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource(new PathPatternParser());
        source.registerCorsConfiguration("/**", config);

        return new CorsWebFilter(source);
    }
}

    /**
     * 过滤器核心方法，为请求添加Client-Address、requestId、requestDate头部信息
     * @param exchange 当前请求上下文，包含请求和响应信息
     * @param chain 网关过滤器链，用于传递给下一个过滤器
     * @return Mono<Void> 异步处理结果
     */
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        //向headers中放文件，记得build
        IdWorker idWorker = new IdWorker();
        ServerHttpRequest host = exchange.getRequest().mutate().header("Client-Address", exchange.getRequest().getRemoteAddress().getAddress().getHostAddress())
            .header("requestId", String.valueOf(idWorker.nextId())).header("requestDate", String.valueOf(System.currentTimeMillis())).build();
        //将现在的request 变成 change对象
        ServerWebExchange build = exchange.mutate().request(host).build();
        return chain.filter(build);
    }

    /**
     * 获取过滤器执行顺序，值越小优先级越高
     * @return 执行顺序值，1表示在日志过滤器之后执行
     */
    @Override
    public int getOrder() {
        return 1;
    }
}

    /**
     * 网关过滤核心方法，拦截所有请求并记录访问日志
     * @param exchange 当前请求上下文，包含请求和响应信息
     * @param chain 网关过滤器链，用于传递给下一个过滤器
     * @return Mono<Void> 异步处理结果
     */
    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {

        ServerHttpRequest request = exchange.getRequest();

        // 获取请求路径（去除上下文路径）
        String requestPath = request.getPath().pathWithinApplication().value();

        // 获取路由配置信息
        Route route = getGatewayRoute(exchange);

        // 获取客户端IP地址
        InetSocketAddress remoteAddress = request.getRemoteAddress();
        String ipAddress = remoteAddress != null ? remoteAddress.getAddress().getHostAddress() : null;

        // 构建日志对象
        Map<String, Object> gatewayLog = new HashMap<>();
        gatewayLog.put("agreement", request.getURI().getScheme());
        gatewayLog.put("requestMethod", request.getMethodValue());
        gatewayLog.put("requestPath", requestPath);
        gatewayLog.put("targetServer", route.getUri().toString());
        gatewayLog.put("startTime", System.currentTimeMillis()); // 使用时间戳保证精度
        gatewayLog.put("ip", ipAddress);
        gatewayLog.put("routeConfig", route); // 路由配置信息

        // 记录请求头
        Map<String, Object> headers = new HashMap<>();
        for (String key : request.getHeaders().keySet()) {
            headers.put(key, request.getHeaders().getFirst(key));
        }
        gatewayLog.put("headers", headers);

        // 获取请求内容类型
        MediaType mediaType = request.getHeaders().getContentType();
        if (mediaType != null) {
            gatewayLog.put("requestContentType", mediaType.toString());
        }

        // 根据内容类型选择处理方式

@Configuration
public class MainDataSourceConfiguration {

    /**
     * 创建主数据源的SqlSessionFactory，扫描classpath下所有mappers目录中的XML映射文件
     * @param dataSource 主数据源，通过@Qualifier("mainSource")注入
     * @return SqlSessionFactory实例
     * @throws Exception 配置异常
     */
    @Bean(name = "mainSqlSessionFactory")
    @Primary
    public SqlSessionFactory sqlSessionFactory(@Qualifier("mainSource") DataSource dataSource) throws Exception {
        SqlSessionFactoryBean bean = new SqlSessionFactoryBean();
        bean.setDataSource(dataSource);
        bean.setMapperLocations(new PathMatchingResourcePatternResolver().getResources("classpath*:mappers/**/*.xml"));
        return bean.getObject();
    }
